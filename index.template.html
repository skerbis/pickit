‚úÖ Confirm Date Plugin

Add a confirmation button to the calendar

  plugins: [new confirmDatePlugin({
    confirmText: "OK",
    showAlways: false,
    theme: "light"
  })]
});

flatpickr("#confirm-datetime", {
  enableTime: true,
  plugins: [new confirmDatePlugin({
    confirmText: "Confirm",
    showAlways: true
  })]
});

‚öôÔ∏è Advanced Options

Additional configuration options

// Inline calendar
flatpickr("#inline-calendar", {
  inline: true
});

// Week numbers
flatpickr("#week-numbers", {
  weekNumbers: true
});

// Disable specific dates
flatpickr("#disabled-dates", {
  disable: [
    "2025-10-15",
    "2025-10-20",
    function(date) {
      // Disable weekends
      return (date.getDay() === 0 || date.getDay() === 6);
    }
  ]
});
